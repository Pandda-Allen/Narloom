    "user_id": "",                  # 用户ID(对应user数据库中的user_id字段)

    "name": "",                     # 资产名称
    "description": "",              # 资产描述

    "created_at": "",               # 资产创建时间
    "updated_at": "",               # 资产更新时间
    === character类型参数 ===
    "role": "",                     # 角色
    traits: {
        "trait1": "trait1",         # 特征
        "trait2": "trait2"          # 特征
    },
    "factions": "",                 # 角色所属阵营
    "group": "",                    # 角色所属分组
    relationships: {
        "name": "relation"          # 关系
    },
    === world类型参数 ===
    "location": "",                 # 地点
    # 剩余参数TBD
}
